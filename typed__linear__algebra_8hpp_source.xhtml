153 template <tla::arithmetic Type>
154 requires tla::singleton<typed_matrix>
155 explicit inline constexpr typed_matrix(const Type &value) {
156 data(0, 0) = tla::element_traits<underlying, Type>::to_underlying(value);

160 template <typename Type>
161 inline constexpr explicit typed_matrix(
162 std::initializer_list<std::initializer_list<Type>> row_list)
163 requires tla::uniform<typed_matrix>
164 {
165 for (std::size_t i{0}; const auto &row : row_list) {
166 for (std::size_t j{0}; const auto &value : row) {
167 data(i, j) =
168 tla::element_traits<underlying, Type>::to_underlying(value);
169 ++j;
170 }
171 ++i;
172 }
173 }

178 template <typename... Types>
179 requires tla::row<typed_matrix> && (not tla::column<typed_matrix>) &&
180 tla::same_size<ColumnIndexes, std::tuple<Types...>>
181 explicit inline constexpr typed_matrix(const Types &...values) {
182 std::tuple value_pack{values...};
183 tla::for_constexpr<0, tla::size<ColumnIndexes>, 1>(
184 [this, &value_pack](auto position) {
185 auto value{std::get<position>(value_pack)};
186 using type = std::remove_cvref_t<decltype(value)>;
187 data[position] =
188 tla::element_traits<underlying, type>::to_underlying(value);
189 });
190 }

383template <typename Matrix, typename RowIndexes, typename ColumnIndexes,
384 typename Char>
385struct std::formatter<fcarouge::typed_matrix<Matrix, RowIndexes, ColumnIndexes>,
386 Char> {
387 constexpr auto parse(std::basic_format_parse_context<Char> &parse_context) {
388 return parse_context.begin();
389 }

391 template <typename OutputIterator>
392 constexpr auto
393 format(const fcarouge::typed_matrix<Matrix, RowIndexes, ColumnIndexes> &value,
394 std::basic_format_context<OutputIterator, Char> &format_context) const
395 -> OutputIterator {
396 format_context.advance_to(std::format_to(format_context.out(), "["));
397
398 for (std::size_t i{0}; i < fcarouge::tla::size<RowIndexes>; ++i) {
399 if (i > 0) {
400 format_context.advance_to(std::format_to(format_context.out(), ", "));
401 }
402
403 format_context.advance_to(std::format_to(format_context.out(), "["));
404
405 for (std::size_t j{0}; j < fcarouge::tla::size<ColumnIndexes>; ++j) {
406 if (j > 0) {
407 format_context.advance_to(std::format_to(format_context.out(), ", "));
408 }
409
410 format_context.advance_to(
411 std::format_to(format_context.out(), "{}", value.data(i, j)));
412 }
413
414 format_context.advance_to(std::format_to(format_context.out(), "]"));
415 }
416
417 format_context.advance_to(std::format_to(format_context.out(), "]"));
418
419 return format_context.out();
420 }
