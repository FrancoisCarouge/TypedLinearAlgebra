58template <typename Tuple, std::size_t... Indices>
59struct lookup<Tuple, std::index_sequence<Indices...>> {
60 using result = typename std::tuple_element<0, Tuple>::type &;
61 using pointer = result (*)(Tuple &) noexcept;

63 static constexpr pointer table[std::tuple_size<Tuple>::value] = {

73template <typename Tuple>
74constexpr typename std::tuple_element<
75 0, typename std::remove_reference<Tuple>::type>::type &
76get(std::size_t index, Tuple &&tuple) {
77 using tuple_type = typename std::remove_reference<Tuple>::type;
78
79 if (index >= std::tuple_size<tuple_type>::value) {
80 throw std::runtime_error("Out of range");
81 }
82
83 return lookup<tuple_type>::table[index](tuple);
84}

86template <typename Tuple, typename Type> struct accessor {

106template <auto Size> void bench() {
107 using tuple =
108 fcarouge::typed_linear_algebra_internal::tuple_n_type<double,
109 Size * Size>;
110 using mdspan = std::mdspan<double, std::extents<std::size_t, Size, Size>,
111 Kokkos::layout_right, accessor<tuple, double>>;
112 tuple storage_a;
113 tuple storage_b;
114 tuple storage_r;
115 mdspan a{&storage_a};
116 mdspan b{&storage_b};
117 mdspan r{&storage_r};
118 std::random_device device;
119 std::mt19937 generator{device()};
120 std::uniform_real_distribution<> distribution{0., 1.};
121
122 for (std::size_t i{0}; i < Size; ++i) {
123 for (std::size_t j{0}; j < Size; ++j) {
124 a(i, j) = distribution(generator);
125 b(i, j) = distribution(generator);
126 }
127 }
