75template <typename Extents>
76constexpr std::size_t extents_size{[] {
77 std::size_t size{1};
78 Extents extents{};
79 for (std::size_t i{0}; i < Extents::rank(); ++i) {
80 size *= extents.extent(i);
81 }
82 return size;
83}()};
