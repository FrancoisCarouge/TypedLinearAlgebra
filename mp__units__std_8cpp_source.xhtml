60// Teach the typed linear algebra library how to convert Eigen' underlying
61// scalar types to and from mp-units' types.
62template <typename To, mp_units::Quantity From>
63struct element_caster<To, From> {
64 [[nodiscard]] constexpr To operator()(From value) const {
65 return value.numerical_value_in(value.unit);
66 }
67};

69template <mp_units::Quantity To, typename From>
70struct element_caster<To, From> {
71 [[nodiscard]] constexpr To operator()(From value) const {
72 return value * To::reference;
73 }
74};

76template <mp_units::Quantity To, typename From>
77struct element_caster<To &, From &> {

90// Set up heterogenously unit typed linear algebra types.
91using representation = double;
92
93template <auto QuantityReference>
94using quantity = mp_units::quantity<QuantityReference, representation>;
95
96template <mp_units::Reference auto QuantityReference>
97using quantity_point =
98 mp_units::quantity_point<QuantityReference,
99 mp_units::default_point_origin(QuantityReference),
100 representation>;
101
102template <typename RowIndexes, typename ColumnIndexes>
103using matrix = typed_matrix<
104 std::mdspan<representation,
105 std::extents<std::size_t, std::tuple_size_v<RowIndexes>,
106 std::tuple_size_v<ColumnIndexes>>>,
107 RowIndexes, ColumnIndexes>;
108
109template <typename... Types>
110using column_vector = typed_column_vector<
111 std::mdspan<representation, std::extents<std::size_t, sizeof...(Types), 1>>,
112 Types...>;
113
114template <typename... Types>
115using row_vector = typed_row_vector<
116 std::mdspan<representation, std::extents<std::size_t, 1, sizeof...(Types)>>,
117 Types...>;
118
119template <std::size_t Rows>
120using column_extents = std::extents<std::size_t, Rows, 1>;
121
122template <std::size_t Columns>
123using row_extents = std::extents<std::size_t, 1, Columns>;
124
125template <typename Extents>
126constexpr std::size_t extents_size{[] {
127 std::size_t size{1};
128 Extents extents{};
129 for (std::size_t i{0}; i < extents.rank(); ++i) {
130 size *= extents.extent(i);
131 }
132 return size;
133}()};

139[[maybe_unused]] auto sample{[] {
140 using mp_units::si::unit_symbols::m;
141 using mp_units::si::unit_symbols::m2;
142 using mp_units::si::unit_symbols::s;
143 using mp_units::si::unit_symbols::s2;
144 using mp_units::si::unit_symbols::s3;
145 using mp_units::one;
146 using mp_units::si::unit_symbols::A;
147 using mp_units::si::unit_symbols::mol;
148 using position = quantity<mp_units::isq::length[m]>;
149 using velocity = quantity<mp_units::isq::velocity[m / s]>;
150 using acceleration = quantity<mp_units::isq::acceleration[m / s2]>;
151 using state = column_vector<position, velocity, acceleration>;
152
153 std::vector v0(extents_size<column_extents<3>>, representation{});
154 std::mdspan s0{v0.data(), column_extents<3>{}};
155 state x0{s0};
156
157 // Elements asignment.
158 x0.at<0>() = 3. * m;
159 x0.at<1>() = 2. * m / s;
160 x0.at<2>() = 1. * m / s2;
161
162 // Printable.
163 std::println("x0 = {}", x0);
164 assert(std::format("{}", x0) == "[[3 m], [2 m/s], [1 m/s²]]");
165
166 // Element assignment and access.
167 x0.at<1>() = 2.5 * m / s;
168 auto x0_1{x0.at<1>()};
169 assert(x0_1 == 2.5 * m / s);
170 assert(std::format("{}", x0_1) == "2.5 m/s");
171
172 // Multiplication with a scalar factor.
173 scale(3., x0);
174 assert(std::format("{}", x0) == "[[9 m], [7.5 m/s], [3 m/s²]]");
175
176 std::vector v4(extents_size<column_extents<3>>, representation{});
177 std::mdspan s4{v4.data(), column_extents<3>{}};
178 state x4{s4};
179
180 // Additions of two vectors of the same types.
181 add(x0, x0, x4);
182 assert(std::format("{}", x4) == "[[18 m], [15 m/s], [6 m/s²]]");
183
184 using state_transpose = row_vector<position, velocity, acceleration>;
185
186 std::vector v5(extents_size<row_extents<3>>, representation{});
187 std::mdspan s5{v5.data(), row_extents<3>{}};
188 state_transpose xt5{s5};
189 xt5.at<0>() = 3. * m;
190 xt5.at<1>() = 2. * m / s;
191 xt5.at<2>() = 1. * m / s2;
192
193 using estimate_uncertainty =
194 matrix<std::tuple<position, velocity, acceleration>,
195 std::tuple<position, velocity, acceleration>>;
196 std::vector v6(extents_size<std::extents<std::size_t, 3, 3>>,
197 representation{});
198 std::mdspan s6{v6.data(), std::extents<std::size_t, 3, 3>{}};
199 estimate_uncertainty p6{s6};
200
201 // Column-row matrix product.
202 matrix_product(x0, xt5, p6);
203 assert(std::format("{}", p6) == "[[27 m², 18 m²/s, 9 m²/s²]," //
204 " [22.5 m²/s, 15 m²/s², 7.5 m²/s³]," //
205 " [9 m²/s², 6 m²/s³, 3 m²/s⁴]]");
206
207 return 0;
208}()};
