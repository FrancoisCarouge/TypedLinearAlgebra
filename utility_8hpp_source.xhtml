158template <std::size_t Begin, std::size_t End, std::size_t Increment,
159 typename Function>
160constexpr void for_constexpr(Function &&function) {
161 if constexpr (Begin < End) {
162 function(std::integral_constant<std::size_t, Begin>());
163 for_constexpr<Begin + Increment, End, Increment>(
164 std::forward<Function>(function));
165 }

211template <typename Type>
212concept is_uniform_typed_matrix =
213 is_typed_matrix<Type> and ([]() {
214 bool result{true};
215
216 for_constexpr<0, std::remove_cvref_t<Type>::rows, 1>([&result](auto i) {
217 for_constexpr<0, std::remove_cvref_t<Type>::columns, 1>([&result,
218 &i](auto j) {
219 result &= std::is_same_v<element<Type, i, j>, element<Type, 0, 0>>;
220 });
221 });
222
223 return result;
224 }());
