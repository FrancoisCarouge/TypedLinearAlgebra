103template <fcarouge::eigen::is_eigen Numerator,
104 fcarouge::eigen::is_eigen Denominator>
105constexpr auto operator/(const Numerator &lhs, const Denominator &rhs)
106 -> fcarouge::eigen::matrix<typename Denominator::Scalar,
107 Numerator::RowsAtCompileTime,
108 Denominator::RowsAtCompileTime> {
109 return rhs.transpose()
110 .fullPivHouseholderQr()
111 .solve(lhs.transpose())
112 .transpose();
113}
114
121template <fcarouge::eigen::is_eigen Denominator>
122constexpr auto operator/(const typename Denominator::Scalar &lhs,
123 const Denominator &rhs)
124 -> fcarouge::eigen::matrix<typename Denominator::Scalar, 1,
125 Denominator::RowsAtCompileTime> {
126 return rhs.transpose()
127 .fullPivHouseholderQr()
128 .solve(fcarouge::eigen::matrix<typename Denominator::Scalar, 1, 1>{lhs})
129 .transpose();
130}
